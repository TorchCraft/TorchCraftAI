27 /// Approximation of Euclidian distance
28 /// This is the same approximation that StarCraft's engine uses
29 /// and thus should be more accurate than true Euclidian distance
30 inline unsigned int disthelper(unsigned int dx, unsigned int dy) {
31  // Helper takes and returns pixels
32  if (dx < dy) {
33  std::swap(dx, dy);
34  }
35  if (dx / 4u < dy) {
36  dx = dx - dx / 16u + dy * 3u / 8u - dx / 64u + dy * 3u / 256u;
37  }
38  return dx;
39 }

48 /// Walktile distance
49 inline float distance(int x1, int y1, int x2, int y2) {
50  unsigned int dx = std::abs(x1 - x2) * unsigned(tc::BW::XYPixelsPerWalktile);
51  unsigned int dy = std::abs(y1 - y2) * unsigned(tc::BW::XYPixelsPerWalktile);
52  return float(disthelper(dx, dy)) / tc::BW::XYPixelsPerWalktile;
53 }

75 /// Distance between two bounding boxes, in pixels.
76 /// Brood War uses bounding boxes for both collisions and range checks
77 inline int pxDistanceBB(
78  int xminA,
79  int yminA,
80  int xmaxA,
81  int ymaxA,
82  int xminB,
83  int yminB,
84  int xmaxB,
85  int ymaxB) {
86  if (xmaxB < xminA) { // To the left
87  if (ymaxB < yminA) { // Fully above
88  return pxdistance(xmaxB, ymaxB, xminA, yminA);
89  } else if (yminB > ymaxA) { // Fully below
90  return pxdistance(xmaxB, yminB, xminA, ymaxA);
91  } else { // Adjecent
92  return xminA - xmaxB;
93  }
94  } else if (xminB > xmaxA) { // To the right
95  if (ymaxB < yminA) { // Fully above
96  return pxdistance(xminB, ymaxB, xmaxA, yminA);
97  } else if (yminB > ymaxA) { // Fully below
98  return pxdistance(xminB, yminB, xmaxA, ymaxA);
99  } else { // Adjecent
100  return xminB - xmaxA;
101  }
102  } else if (ymaxB < yminA) { // Above
103  return yminA - ymaxB;
104  } else if (yminB > ymaxA) { // Below
105  return yminB - ymaxA;
106  }
107 
108  return 0;
109 }

161 // Get movement towards position p, rotated by angle in degrees.
162 // If not exact, we click past it so we maintain flyer acceleration
163 // Positive angle rotates from the top right to the bottom left corner,
164 // since the y axis points down.
165 inline Position getMovePosHelper(
166  int ux,
167  int uy,
168  int px,
169  int py,
170  int mx,
171  int my,
172  double angle,
173  bool exact) {
174  auto fdirX = px - ux;
175  auto fdirY = py - uy;
176  if (fdirX == 0 && fdirY == 0) {
177  return Position(px, py);
178  }
179  auto rad = angle * kDegPerRad;
180  auto c = std::cos(rad);
181  auto s = std::sin(rad);
182  auto dirX = fdirX * c - fdirY * s;
183  auto dirY = fdirX * s + fdirY * c;
184  if (!exact && dirX * dirX + dirY * dirY < 10) {
185  // Approximate, I don't want to compute the magnitude
186  // Clicks at least 10 walktiles ahead
187  auto div = std::abs(dirX == 0 ? dirY : dirX);
188  dirX = dirX / div * 10;
189  dirY = dirY / div * 10;
190  }
191  return Position(
192  utils::clamp(ux + (int)dirX, 0, mx - 1),
193  utils::clamp(uy + (int)dirY, 0, my - 1));
194 }

283 inline bool prerequisitesReady(State* state, const BuildType* buildType) {
284  auto& unitsInfo = state->unitsInfo();
285  for (auto* prereq : buildType->prerequisites) {
286  if (prereq->isUnit()) {
287  bool hasPrereq = !unitsInfo.myCompletedUnitsOfType(prereq).empty();
288  if (!hasPrereq) {
289  if (prereq == buildtypes::Zerg_Spire) {
290  hasPrereq =
291  !unitsInfo.myUnitsOfType(buildtypes::Zerg_Greater_Spire).empty();
292  } else if (prereq == buildtypes::Zerg_Hatchery) {
293  hasPrereq =
294  !unitsInfo.myCompletedUnitsOfType(buildtypes::Zerg_Lair).empty();
295  if (!hasPrereq) {
296  hasPrereq = !unitsInfo.myUnitsOfType(buildtypes::Zerg_Hive).empty();
297  }
298  } else if (prereq == buildtypes::Zerg_Lair) {
299  hasPrereq = !unitsInfo.myUnitsOfType(buildtypes::Zerg_Hive).empty();
300  }
301  }
302  if (!hasPrereq) {
303  return false;
304  }
305  } else if (prereq->isUpgrade()) {
306  if (state->getUpgradeLevel(prereq) < prereq->level) {
307  return false;
308  }
309  } else if (prereq->isTech()) {
310  if (!state->hasResearched(prereq)) {
311  return false;
312  }
313  } else {
314  VLOG(2) << "Unknown prerequisite " << buildTypeString(prereq) << " for "
315  << buildTypeString(buildType);
316  return false;
317  }
318  }
319  return true;
320 }
