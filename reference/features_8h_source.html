34  * Defines a family of "plain" features.

36  * These features can directly be extracted from the bot State into a spatial
37  * FeatureData instance. Use featurizePlain() to extract these features.

39 enum class PlainFeatureType {
40  Invalid = -1,
41  /// Ground height: 0 (low), 1 (high) or 2 (very high); -1 outside of map.
42  GroundHeight = 1,
43  /// Whether units can walk here or not: 0 or 1; -1 outside of map
44  Walkability,
45  /// Whether buildings can be placed here or not: 0 or 1; -1 outside of map
46  Buildability,
47  /// Whether this position is under the fog of war: 0 or 1; -1 outside of map
48  FogOfWar,
49  /// Whether there is creep here: 0 or 1; -1 outside of map

138  template <typename Archive>
139  void save(Archive& ar) const {
140  int kind = -1;
141  int value = -1;
142  type.match(
143  [&](PlainFeatureType t) {
144  kind = kPlainType;
145  value = static_cast<int>(t);
146  },
147  [&](CustomFeatureType t) {
148  kind = kCustomType;
149  value = static_cast<int>(t);
150  });
151  ar(CEREAL_NVP(kind),
152  CEREAL_NVP(value),
153  CEREAL_NVP(name),
154  CEREAL_NVP(numChannels));
155  }
156 
157  template <typename Archive>
158  void load(Archive& ar) {
159  int kind = -1;
160  int value = -1;
161  ar(CEREAL_NVP(kind),
162  CEREAL_NVP(value),
163  CEREAL_NVP(name),
164  CEREAL_NVP(numChannels));
165  switch (kind) {
166  case kPlainType:
167  type = static_cast<PlainFeatureType>(value);
168  break;
169  case kCustomType:
170  type = static_cast<CustomFeatureType>(value);
171  break;
172  default:
173  throw std::runtime_error(
174  "Unknown feature kind: " + std::to_string(kind));
175  };
176  }

253 /**
254  * Maps walktile positions to feature positions for a given bounding box.
255  *
256  * This is mostly useful for actual featurizer implementations. Use `(x, y)` to
257  * map a position. For invalid positions (outside of the intersection of
258  * bounding box and map rectangle), `(-1, -1)` is returned.
259  */
260 struct FeaturePositionMapper {
261  FeaturePositionMapper(Rect const& boundingBox, Rect const& mapRect) {
262  Rect ir = boundingBox.intersected(mapRect);
263  irx1 = ir.left();
264  iry1 = ir.top();
265  irx2 = ir.right() - 1;
266  iry2 = ir.bottom() - 1;
267  offx = mapRect.x - boundingBox.x;
268  offy = mapRect.y - boundingBox.y;
269  }
270  int irx1;
271  int irx2;
272  int iry1;
273  int iry2;
274  int offx;
275  int offy;
276 
277  Position operator()(Position const& pos) const {
278  if (pos.x < irx1 || pos.y < iry1 || pos.x > irx2 || pos.y > iry2) {
279  return kInvalidPosition;
280  }
281  return Position(pos.x + offx, pos.y + offy);
282  }
283 };
