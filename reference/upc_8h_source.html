27 /**
28  * (Unit, Position, Command) tuple.
29  * Specifies the (who, where, what) of an action
30  *
31  * UPCTuples are used for module-to-module communication (via the Blackboard).
32  * Posting a UPCTuple to the Blackboard is akin to requesting a specific action
33  * from another module, and consuming a UPCTuple from the Blackboard implies
34  * that the consumer implements this action. The implementation can also consist
35  * of refining the UPCTuple so that a more lower-level module can actually
36  * execute it.
37  *
38  * For example, a build order module might post a UPCTuple to create a certain
39  * unit type, and a builder module might wait until their are sufficient
40  * resources before consuming it and then select a worker and a location.
41  *
42  * The UPCToCommandModule translates executable UPCTuples (ones with
43  * sharp unit, position and command entries) to actual game commands.
44  */
45 struct UPCTuple {
46  /// An empty default item for the variants defined below
47  using Empty = char; // Any better options here?
48  using UnitMap = std::unordered_map<Unit*, float>;
49  using CommandMap = std::unordered_map<Command, float>;
50  using BuildTypeMap = std::unordered_map<BuildType const*, float>;
51

54  // Using variant for some types but not others is not very consistent, but it
55  // simplifies code working with UPCTuples.
56 
57  /// A typedef for the unit distribution.
58  using UnitT = UnitMap;
59 #ifdef HAVE_TORCH
60  /// A typedef for the position distribution.
61  /// Possible values:
62  /// - Empty: unspecified position, i.e. uniform over the entire map
63  /// - Position: A single x/y position with probability of one.
64  /// This is more efficient then having a one-hot position tensor.
65  /// - Area*: An area so that every position in this area has probability
66  /// 1/(walktiles in area), and every position outside has probability 0.
67  /// - UnitMap: A distribution over units which can be used instead of
68  /// position. Sometimes it is more efficient to specify target units
69  /// directly (e.g. for resource mining or for attacks).
70  /// - torch::Tensor: A distribution over the whole map. First dimension is Y,
71  /// second is X.
72  using PositionT =
73  mapbox::util::variant<Empty, Position, Area*, UnitMap, torch::Tensor>;
74 #else // HAVE_TORCH
75  using PositionT = mapbox::util::variant<Empty, Position, Area*, UnitMap>;
76 #endif // HAVE_TORCH
77  /// A typedef for the command distribution
78  using CommandT = CommandMap;
79 #ifdef HAVE_TORCH
80  /// A typedef for additional structured information ("state").
81  /// Possible values:
82  /// - Empty: unspecified state
83  /// - BuildTypeMap: A distribution over unit types. This is useful for UPCs
84  /// with the "Create" command.
85  /// - std::string: An arbitrary string.
86  /// - Position: An arbitrary position.
87  /// - torch::Tensor: An arbitrary tensor.
88  using StateT = mapbox::util::variant<Empty,
89  BuildTypeMap,
90  std::string,
91  Position,
92  SetCreatePriorityState,
93  torch::Tensor>;
94 #else // HAVE_TORCH
95  using StateT = mapbox::util::variant<Empty,
96  BuildTypeMap,
97  std::string,
98  Position,
99  SetCreatePriorityState>;
100 #endif // HAVE_TORCH
