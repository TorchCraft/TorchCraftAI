27 /**
28  * A multi-threaded iterator that performs decerealization of objects and
29  * returns data in batches.
30  *
31  * Here, batches means instances of std::vector<T>.
32  *
33  * The iterator will read data from all files specified by the list of paths
34  * passed to the constructor *once*. The order of files is retained and will be
35  * reflected in the resulting batches.
36  *
37  * Call `next()` to retrieve the next batch of data. It will block until enough
38  * data is available to form a batch with the requested size. If the iterator
39  * cannot advance anymore (`hasNext()` returns false), the call with throw an
40  * exception. The last batch returned by this function may have less that
41  * `batchSize` elements or be empty, depending on the actual data.
42  *
43  * For data files that cannot be decerealized (e.g. because the file cannot be
44  * accessed, or because decerialization failed), the iterator will print a
45  * message via glog but otherwise resume operation as usual.
46  *
47  * You will probably want to use this via DataReader<T>.
48  */

158  * cerealized versions of T. zstd decompression will transparently work. The

277 DataReaderIterator<T>::~DataReaderIterator() {
278  {
279  // Move iterator to end and clear results queue so that threads can
280  // finish their current operation.

299 template <typename T>
300 std::vector<T> DataReaderIterator<T>::next() {
301  std::unique_lock<std::mutex> lock(mutex_);
302  if (dataQueue_.empty() && threadPos_ >= paths_.size()) {
303  throw std::runtime_error("Data iterator is already at end");
304  }
305 
306  std::vector<T> batch;
307  while (pos_ < paths_.size() && batch.size() < batchSize_) {
308  auto curPos = pos_++;
309 
310  int numAttempts = 0;
311  auto it = dataQueue_.end();
312 
313  consumerCV_.wait(lock, [&] {
314  it = dataQueue_.find(curPos);
315  if (it == dataQueue_.end()) {
316  // The requested datum is not in the queue yet -- wait a while.
317  // If we waited for a while already, increase queue size and hope
318  // that we'll eventually get our requested datum's future.
319  if ((++numAttempts % 5) == 0) {
320  maxQueueSize_ *= 1.5;
321  }
322  return false;
323  }
324 
325  return true;
326  });
327 
328  try {
329  batch.push_back(it->second.get());
330  } catch (std::exception const& e) {
331  LOG(WARNING) << "Cannot query result for datum " << pos_ << ", skipping ("
332  << e.what() << ")";
333  }
334  dataQueue_.erase(it);
335  }
336 
337  // Back to normal
338  maxQueueSize_ = kMaxBatchesInQueue * batchSize_;
339 
340  prodCV_.notify_all();
341  return batch;
342 }

344 template <typename T>
345 void DataReaderIterator<T>::read() {
346  init_();
347 
348  while (true) {
349  size_t curPos;
350  std::promise<T> dataPromise;
351  bool done = false;
352 
353  { // Critical section interacting with producer and consumer queue
354  std::unique_lock<std::mutex> lock(mutex_);
355  while (true) {
356  if (threadPos_ >= paths_.size()) {
357  done = true;
358  break;
359  }
360  if (dataQueue_.size() < maxQueueSize_) {
361  curPos = threadPos_++;
362  dataQueue_[curPos] = dataPromise.get_future();
363  break;
364  }
365 
366  // No space in queue, return to waiting. In case the specific datum in
367  // question is blocked here, we notify the consumer (next()) to
368  // provide them a chance for increasing the queue size
369  consumerCV_.notify_one();
370 
371  prodCV_.wait(lock);
372  }
373  }
374  if (done) {
375  break;
376  }
377 
378  std::string filePath;
379  auto const& curPath = paths_[curPos];
380  if (!prefix_.empty() && !curPath.empty() && curPath[0] != '/') {
381  filePath = prefix_ + "/" + curPath;
382  } else {
383  filePath = curPath;
384  }
385 
386  // Read data, fulfill promise from above
387  try {
388  VLOG(4) << "Reading data from " << filePath;
389  zstd::ifstream is(filePath);
390  cereal::BinaryInputArchive archive(is);
391  T d;
392  archive(d);
393  dataPromise.set_value(std::move(d));
394  } catch (std::exception const& e) {
395  VLOG(0) << "Invalid data file " << filePath << ", skipping (" << e.what()
396  << ")";
397  try {
398  dataPromise.set_exception(std::current_exception());
399  } catch (std::exception const& e2) {
400  LOG(ERROR) << "Cannot propagate exception: " << e2.what();
401  }
402  }
403 
404  // Use notify_all() for the (unlikely) case where multiple threads are
405  // waiting in next().
406  consumerCV_.notify_all();
407  }
408 
409  std::unique_lock<std::mutex> lock(mutex_);
410  threadsDone_.insert(std::this_thread::get_id());
411 }

429 template <typename T, typename F>
430 typename DataReaderTransform<T, F>::Result DataReaderTransform<T, F>::next() {
431  std::unique_lock<std::mutex> lock(mutex_);
432  if (queue_.empty() && done_) {
433  throw std::runtime_error("Data iterator is already at end");
434  }
435 
436  consumerCV_.wait(lock, [&] { return !queue_.empty(); });
437  auto result = std::move(queue_.front());
438  queue_.pop();
439  prodCV_.notify_all();
440  return result;
441 }

443 template <typename T, typename F>
444 void DataReaderTransform<T, F>::run() {
445  init_();
446 
447  std::unique_lock<std::mutex> lock(mutex_);
448  // We want this check to be locked so that when we are at the end of the
449  // iterator, we do not leave the critical section before setting done_ to
450  // true.
451  while (it_->hasNext()) {
452  lock.unlock();
453  auto result = fn_(it_->next());
454  lock.lock();
455 
456  prodCV_.wait(
457  lock, [&] { return done_ || queue_.size() < kMaxResultsInQueue; });
458  if (done_) {
459  break;
460  }
461 
462  queue_.push(std::move(result));
463  consumerCV_.notify_one();
464  }
465  done_ = true;
466 }
