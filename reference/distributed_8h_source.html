30 /**
31  * This is a wrapper around ProcessGroup::Work.
32  * We wait for the work to be finished on destruction, to ensure transfer
33  * completes. Additionally, this subclass provides support for waiting on
34  * multiple pieces of work to finish, in the case of syncing entire models.
35  *
36  * The comments for this class are copied from ProcessGroup::Work and may be out
37  * of date when PyTorch updates...
38  **/

40 class Work {
41  public:
42  Work(std::function<void()> onFinish);
43  Work(Work const&) = delete;
44  Work(Work&&);
45  ~Work();

47  // Checks if request has completed. Non-blocking operation.
48  bool isCompleted();
49 
50  // Returns if the work completed successfully.
51  // If false, the exception function can be called to get details.
52  bool isSuccess();
53 
54  // Ensures that operations on the output tensors that are invoked
55  // after this function returns are correctly sequenced after the
56  // asynchronous completion of this work.
57  //
58  // For CUDA tensors, it inserts stream synchronization such that
59  // the streams of the caller wait for completion of the
60  // asynchronous operations on the destination tensors.
61  //
62  // For CPU tensors, it is currently a nop.
63  //
64  // This function should only be used if the caller polls for
65  // completion through the `isCompleted` function, it has returned
66  // true, and the `isSuccess` function also has returned true.
67  //
68  void synchronize();
69 
70  // Waits until request completes. Blocking operation.
71  // Returns false if the work completed with an exception.
72  //
73  // Functionally equivalent to:
74  //
75  // while (!isCompleted()) { /* nop */ }
76  // auto success = isSuccess();
77  // if (success) { synchronize(); }
78  // return success;
79  //
80  bool wait();
81 
82  // Returns exception if wait() returned false.
83  // This will return the first exception encountered.
84  const std::exception& exception() const;
85 
86  private:
87  Work() = default;
88  Work(std::vector<std::shared_ptr<ProcessGroup::Work>>&&);
89  void add(std::shared_ptr<ProcessGroup::Work>);
90  void add(Work&&);

92  std::vector<std::shared_ptr<ProcessGroup::Work>> works_;
93  std::function<void()> onFinish_ = nullptr;
94  friend class Context;
