151  double radians = degrees / kDegPerRad;

269  Rect2T<T> intersected(Rect2T<T> const& r) const {
270  if (empty() || r.empty()) {
271  return Rect2T<T>();
272  }
273 
274  T left1 = x;
275  T right1 = x + w;
276  T left2 = r.x;
277  T right2 = r.x + r.w;
278 
279  if (left1 >= right2 || left2 >= right1) {
280  // No intersection
281  return Rect2T<T>();
282  }
283 
284  T top1 = y;
285  T bottom1 = y + h;
286  T top2 = r.y;
287  T bottom2 = r.y + r.h;
288 
289  if (top1 >= bottom2 || top2 >= bottom1) {
290  // No intersection
291  return Rect2T<T>();
292  }
293 
294  Rect2T<T> t;
295  t.x = std::max(left1, left2);
296  t.y = std::max(top1, top2);
297  t.w = std::min(right1, right2) - t.x;
298  t.h = std::min(bottom1, bottom2) - t.y;
299  return t;
300  }
